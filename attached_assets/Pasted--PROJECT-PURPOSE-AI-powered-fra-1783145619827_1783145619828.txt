# ------------------------------
# BACKEND / API SETUP

# Core endpoints:
# - /api/locations
# - /api/licenses
# - /api/compliance
# - /api/tasks
# - /api/audits
# - /api/documents
# - /api/ai/chat (AI proxy)
pnpm start:dev

# ------------------------------
# DATABASE SETUP
# ------------------------------
cd ../../lib/db
pnpm install
pnpm prisma generate
pnpm prisma migrate dev

# - Audits (location history)
